149 //
150 // Calibration values
151 //-------------------
152 // Three calibration values are used in this sketch: f_powerCal, f_phaseCal and f_voltageCal.
153 // With most hardware, the default values are likely to work fine without
154 // need for change. A compact explanation of each of these values now follows:
155 
156 // When calculating real power, which is what this code does, the individual
157 // conversion rates for voltage and current are not of importance. It is
158 // only the conversion rate for POWER which is important. This is the
159 // product of the individual conversion rates for voltage and current. It
160 // therefore has the units of ADC-steps squared per Watt. Most systems will
161 // have a power conversion rate of around 20 (ADC-steps squared per Watt).
162 //
163 // powerCal is the RECIPR0CAL of the power conversion rate. A good value
164 // to start with is therefore 1/20 = 0.05 (Watts per ADC-step squared)
165 //
166 constexpr float f_powerCal[NO_OF_PHASES]{0.0532f, 0.0542f, 0.0532f};
167 //
168 // f_phaseCal is used to alter the phase of the voltage waveform relative to the current waveform.
169 // The algorithm interpolates between the most recent pair of voltage samples according to the value of f_phaseCal.
170 //
171 // With f_phaseCal = 1, the most recent sample is used.
172 // With f_phaseCal = 0, the previous sample is used
173 // With f_phaseCal = 0.5, the mid-point (average) value in used
174 //
175 // NB. Any tool which determines the optimal value of f_phaseCal must have a similar
176 // scheme for taking sample values as does this sketch.
177 //
178 constexpr float f_phaseCal{1};
179 //
180 // When using integer maths, calibration values that have been supplied in floating point form need to be rescaled.
181 constexpr int16_t i_phaseCal{256};
182 constexpr int16_t p_phaseCal{8};
183 //
184 // For datalogging purposes, f_voltageCal has been added too. Because the range of ADC values is
185 // similar to the actual range of volts, the optimal value for this cal factor is likely to be
186 // close to unity.
187 constexpr float f_voltageCal[NO_OF_PHASES]{0.985f, 1.003f, 0.987f} /*{1.03f, 1.03f, 1.03f}*/;

617 void processCurrentRawSample(const uint8_t phase, const int16_t rawSample)
618 {
619  static int32_t sampleIminusDC;
620  static int32_t phaseShiftedSampleVminusDC;
621  static int32_t filtV_div4;
622  static int32_t filtI_div4;
623  static int32_t instP;
624 
625  // remove most of the DC offset from the current sample (the precise value does not matter)
626  sampleIminusDC = ((int32_t)(rawSample - l_DCoffset_I_nom)) << 8;
627  //
628  // phase-shift the voltage waveform so that it aligns with the grid current waveform
629  phaseShiftedSampleVminusDC = l_lastSampleVminusDC[phase] + (((l_sampleVminusDC[phase] - l_lastSampleVminusDC[phase]) << p_phaseCal) >> 8);
630  //
631  // calculate the "real power" in this sample pair and add to the accumulated sum
632  filtV_div4 = phaseShiftedSampleVminusDC >> 2; // reduce to 16-bits (now x64, or 2^6)
633  filtI_div4 = sampleIminusDC >> 2; // reduce to 16-bits (now x64, or 2^6)
634  instP = filtV_div4 * filtI_div4; // 32-bits (now x4096, or 2^12)
635  instP >>= 12; // scaling is now x1, as for Mk2 (V_ADC x I_ADC)
636 
637  l_sumP[phase] += instP; // cumulative power, scaling as for Mk2 (V_ADC x I_ADC)
638  l_sumP_atSupplyPoint[phase] += instP; // cumulative power, scaling as for Mk2 (V_ADC x I_ADC)
639 }
